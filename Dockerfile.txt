# Use official Jenkins LTS image as the base
FROM jenkins/jenkins:lts

# Install dependencies for SonarScanner
USER root
RUN apt-get update && \
    apt-get install -y openjdk-17-jdk wget unzip

# Install SonarScanner
RUN wget https://binaries.sonarsource.com/Distribution/sonar-scanner-cli/sonar-scanner-cli-4.8.0.2856-linux.zip -O sonar-scanner.zip && \
    unzip sonar-scanner.zip -d /opt && \
    rm sonar-scanner.zip

# Set environment variables for SonarScanner
ENV SONAR_SCANNER_HOME=/opt/sonar-scanner-4.8.0.2856-linux
ENV PATH=$SONAR_SCANNER_HOME/bin:$PATH

# Switch back to the Jenkins user
USER jenkins
